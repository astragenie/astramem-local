```ts
POST /register
body schema (Zod):
  client: string (min 1)         // 'astramem-plugin'
  version: string (semver)
auth: Bearer (existing middleware)
returns: {
  ok: true,
  daemon_version: string,        // package.json version

  capabilities: ['ingest','recall','remember'],
```

  - Win: `%APPDATA%/AstraMemory`
